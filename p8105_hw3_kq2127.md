``` r
pb3_df1 = 
  read_csv(
    file = "./data/accel_data.csv") %>% 
  janitor::clean_names() %>% 
  mutate(
    day_type = case_when(
      day  %in% c ("Saturday", "Sunday") ~ "Weekend",
      day  %in% c ("Monday", "Tuesday", "Wednesday", "Thursday", "Friday") ~ "Weekday",
      TRUE      ~"" 
    )) %>% 
  select(week, day_id, day, day_type, everything())
```
