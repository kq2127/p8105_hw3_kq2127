instacart %>%
  group_by(aisle_id) %>%
  summarize(n_obs = n()) %>% 
  filter(n_obs > 1000) %>% 
  ggplot(aes(x = aisle_id, y = n_obs)) + 
    geom_point(color = "red", alpha = .5) +
  labs(
    title = "Items Ordered per Aisle",
    x = "Aisle ID",
    y = "Items Ordered (#)"
  )+
  scale_y_continuous(
    breaks = c(1000, 50000, 150000)
  )
